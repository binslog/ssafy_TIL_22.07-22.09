### Bubble

```python
a = [8, 3, 12, 10, 1]

for i in range(len(a)-1,0,-1):  # 4 3 2 1
    for j in range (0,i): # i가 4일때 0123  i가 3일때 012...
        if a[j]>a[j+1]:
            a[j],a[j+1]=a[j+1],a[j]

print(a)


for i in range(len(a)): # 0 1 2 3
    for j in range(0,len(a)-2-i): #0 부터 3 2 1 0
        if a[j] > a[j + 1]:
            a[j], a[j + 1] = a[j + 1], a[j]

print(a)
```

> Counting Sort


```python

### Counting Sort
n = int(input())
a = list(map(int,input().split()))

bucket = [0] * 101 # 먼저 bucket부터 만든다.

# dat 등록
for i in range(n):
    bucket[a[i]] += 1 # 버켓만들기
    print(bucket)

# 누적합 구하기
for i in range(1, len(bucket)):
    bucket[i]+=bucket[i-1]
    # bucket[i]=bucket[i]+bucket[i-1]
    print(bucket[i])

# 값 넣기
result=[0] * 101
for i in range(n):
    index=a[i]
    result[bucket[index]-1] = a[i]
    bucket[index]-=1


for i in range(len(a)):
    if bucket[i]>0 :
      print(f'{i}: {bucket}개 존재합니다.')

```





> 잔돈 반환하기

```python
coin=[100,50,10]
change=int(input()) # 거슬러 줄 돈   250
answer=0  # 총 동전 사용 개수

index=0
while(1):
    cnt=change//coin[index]   # cnt=2 
    change-=(cnt*coin[index]) # 나머지 거슬러 줘야 할 돈 50
    answer+=cnt
    index+=1 # 그다음 50원 동전 사용을 하기 위해 index 1 증가
    if index==3:
        break
print(answer)


coin=[100,50,10]
change=int(input()) # 거슬러 줄 돈   250
answer= 0  # 총 동전 사용 개수


for i in range(len(coin)) :    
    if change < coin[-1] :
        break
    else:
        cnt=0
        cnt += change // coin[i]         
        change -= cnt * coin[i]

        answer += cnt

print(answer)
```

```python
board = [
    ["A", "B", "G", "K"],
    ["T", "T", "A", "B"],
    ["A", "C", "T", "T"]
]

def findptn(dy,dx):
    for i in range(2):
        for j in range(2):
            if board[dy+i][dx+j]!=ptn[i][j] :
                return 0
        return 1    

cnt=0

    for x in range(3):
        if findptn(y,x) :# 0,0 ~~ 1,2
```
